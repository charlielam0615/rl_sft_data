\[
\left(x^2 + \frac{1}{x^2}\right)^2 = x^4 + 2 \cdot x^2 \cdot \frac{1}{x^2} + \frac{1}{x^4} = x^4 + 2 + \frac{1}{x^4}.

\[
x^4 + \frac{1}{x^4} = \left(x^2 + \frac{1}{x^2}\right)^2 - 2.

\left(x^2 + \frac{1}{x^2}\right)^2 - 2 = 23.
\]
[/IMP]

\[
