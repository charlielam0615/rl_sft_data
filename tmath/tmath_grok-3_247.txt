\[ \sqrt{(4 - 2a)^2 + ((-1) - (a-4))^2} = 2\sqrt{10} \]

\[ \sqrt{(4 - 2a)^2 + (3 - a)^2} = 2\sqrt{10} \]

\[ (4 - 2a)^2 + (3 - a)^2 = (2\sqrt{10})^2 \]
